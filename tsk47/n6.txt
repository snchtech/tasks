# 1) –û—Ç—Ä–∏–º–∞–Ω–Ω—è —Ä–µ–∂–∏–º—É –µ—Ç–∞–ø—É (0=–ø–∞—Ä–∞–ª–µ–ª—å–Ω–æ, 1=–ø–æ—Å–ª—ñ–¥–æ–≤–Ω–æ)

> –û–¥–∏–Ω –µ—Ç–∞–ø = –±–∞–≥–∞—Ç–æ –ø—ñ–¥–µ—Ç–∞–ø—ñ–≤ (rows —É `SIGNATURERIGHTS`), —Ç–æ–º—É –±–µ—Ä–µ–º–æ –∞–≥—Ä–µ–≥–∞—Ç–æ–º (`min/max`, –æ–±–µ—Ä–∏ —Å–≤—ñ–π –¥–µ—Ñ–æ–ª—Ç). –Ø –∑–∞–ª–∏—à–∏–≤ `min`: —è–∫—â–æ —Ö–æ—á –æ–¥–∏–Ω —Ä—è–¥–æ–∫ –µ—Ç–∞–ø—É 0 ‚Üí –≤–≤–∞–∂–∞—î–º–æ –µ—Ç–∞–ø –ø–∞—Ä–∞–ª–µ–ª—å–Ω–∏–º.

```plsql
-- 0 = –ø–∞—Ä–∞–ª–µ–ª—å–Ω–æ, 1 = –ø–æ—Å–ª—ñ–¥–æ–≤–Ω–æ. –Ø–∫—â–æ –¥–∞–Ω–∏—Ö –Ω–µ–º–∞ ‚Äî 1 (–ø–æ—Å–ª—ñ–¥–æ–≤–Ω–æ) —è–∫ –±–µ–∑–ø–µ—á–Ω–∏–π –¥–µ—Ñ–æ–ª—Ç.
function get_stage_mode(
  p_task_id      in number,
  p_page_list_id in number,
  p_stage_no     in number   -- 2 / 4 / 6
) return number
is
  l_mode number;
begin
  select nvl(min(sr.stage_mode), 1)
    into l_mode
    from TaskTracker.SIGNATURERIGHTS sr
    join TaskTracker.DICT_APP_CONFIRM_LIST d
      on d.id = sr.stages_id
     and d.parent_stage_id is not null
    join TaskTracker.DICT_APP_CONFIRM_LIST p
      on p.id = d.parent_stage_id
   where sr.task_id        = p_task_id
     and p.page_list_id    = p_page_list_id
     and p.parent_stage_id is null
     and p.stage           = p_stage_no;

  return l_mode;
exception
  when no_data_found then
    return 1;
end;
```

# 2) –ü–µ—Ä–µ–≤—ñ—Ä–∫–∞ ¬´–µ—Ç–∞–ø –∑–∞–≤–µ—Ä—à–µ–Ω–∏–π / –∞–∫—Ç–∏–≤–Ω–∏–π¬ª

> –ê–∫—Ç–∏–≤–Ω–∏–π –µ—Ç–∞–ø = **—É—Å—ñ –ø–æ–ø–µ—Ä–µ–¥–Ω—ñ** –µ—Ç–∞–ø–∏ –ø–æ–≤–Ω—ñ—Å—Ç—é –ø—ñ–¥–ø–∏—Å–∞–Ω—ñ **—ñ** –≤ –ø–æ—Ç–æ—á–Ω–æ–º—É —â–µ –ª–∏—à–∏–ª–∏—Å—è –Ω–µ–ø—ñ–¥–ø–∏—Å–∞–Ω—ñ —Ä—è–¥–∫–∏.
> –ü–æ–≤–Ω—ñ—Å—Ç—é –∑–∞–≤–µ—Ä—à–µ–Ω–∏–π = **–∫–æ–∂–µ–Ω** `SIGNATURERIGHTS` –µ—Ç–∞–ø—É –º–∞—î –∑–∞–ø–∏—Å —É `SIGNATUREHISTORY`.

```plsql
-- –∑–∞–≤–µ—Ä—à–µ–Ω–∏–π (—É—Å—ñ SR –µ—Ç–∞–ø—É –º–∞—é—Ç—å —Ö–æ—á –æ–¥–∏–Ω –∑–∞–ø–∏—Å —É SH)
function is_stage_completed(
  p_task_id      in number,
  p_page_list_id in number,
  p_stage_no     in number
) return boolean
is
  l_total number;
  l_signed number;
begin
  -- —Å–∫—ñ–ª—å–∫–∏ SR —É —Ü—å–æ–º—É –µ—Ç–∞–ø—ñ
  select count(*)
    into l_total
    from TaskTracker.SIGNATURERIGHTS sr
    join TaskTracker.DICT_APP_CONFIRM_LIST d
      on d.id = sr.stages_id
     and d.parent_stage_id is not null
    join TaskTracker.DICT_APP_CONFIRM_LIST p
      on p.id = d.parent_stage_id
   where sr.task_id        = p_task_id
     and p.page_list_id    = p_page_list_id
     and p.parent_stage_id is null
     and p.stage           = p_stage_no;

  if l_total = 0 then
    return false; -- –Ω–µ–º–∞—î –∫–æ–≥–æ –ø—ñ–¥–ø–∏—Å—É–≤–∞—Ç–∏ ‚Üí –≤–≤–∞–∂–∞—î–º–æ ¬´–Ω–µ –∑–∞–≤–µ—Ä—à–µ–Ω–æ¬ª
  end if;

  -- —Å–∫—ñ–ª—å–∫–∏ –∑ –Ω–∏—Ö –º–∞—é—Ç—å —Ö–æ—á –æ–¥–∏–Ω –ø—ñ–¥–ø–∏—Å
  select count(distinct sr.id)
    into l_signed
    from TaskTracker.SIGNATURERIGHTS  sr
    join TaskTracker.DICT_APP_CONFIRM_LIST d
      on d.id = sr.stages_id
     and d.parent_stage_id is not null
    join TaskTracker.DICT_APP_CONFIRM_LIST p
      on p.id = d.parent_stage_id
    join TaskTracker.SIGNATUREHISTORY sh
      on sh.signright_id = sr.id
   where sr.task_id        = p_task_id
     and p.page_list_id    = p_page_list_id
     and p.parent_stage_id is null
     and p.stage           = p_stage_no;

  return (l_signed = l_total);
end;

-- –∞–∫—Ç–∏–≤–Ω–∏–π = —É—Å—ñ –ø–æ–ø–µ—Ä–µ–¥–Ω—ñ –∑–∞–≤–µ—Ä—à–µ–Ω—ñ –Ü –ø–æ—Ç–æ—á–Ω–∏–π —â–µ –ù–ï –∑–∞–≤–µ—Ä—à–µ–Ω–∏–π
function is_stage_active(
  p_task_id      in number,
  p_page_list_id in number,
  p_stage_no     in number
) return boolean
is
  l_prev_ok boolean := true;
  l_stg_completed boolean;
begin
  -- —É—Å—ñ –ø–æ–ø–µ—Ä–µ–¥–Ω—ñ (2->–Ω–µ–º–∞ –ø–æ–ø–µ—Ä–µ–¥–Ω—ñ—Ö; 4->–ø–µ—Ä–µ–≤—ñ—Ä—è—î–º–æ 2; 6->–ø–µ—Ä–µ–≤—ñ—Ä—è—î–º–æ 2 —ñ 4)
  if p_stage_no > 2 then
    for s in (select 2 as stg from dual
               union all select 4 from dual
               union all select 6 from dual
              where stg < p_stage_no)
    loop
      if not is_stage_completed(p_task_id, p_page_list_id, s.stg) then
        l_prev_ok := false;
        exit;
      end if;
    end loop;
  end if;

  l_stg_completed := is_stage_completed(p_task_id, p_page_list_id, p_stage_no);

  return (l_prev_ok and not l_stg_completed);
end;
```
