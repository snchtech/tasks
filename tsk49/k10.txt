```js
function sortSubtasksByDate() {
  const ig = apex.region('SUBTASKS_IG').widget();
  const grid = ig.interactiveGrid('getViews', 'grid');
  const model = grid.model;

  const records = [];
  model.forEach(function (record) {
    records.push(record);
  });

  records.sort(function (a, b) {
    const da = model.getValue(a, 'PLANNING_DATE_END');
    const db = model.getValue(b, 'PLANNING_DATE_END');

    if (!da && !db) return 0;
    if (!da) return 1;
    if (!db) return -1;

    return new Date(da) - new Date(db);
  });

  model.clear();
  records.forEach(function (r) {
    model.insertRecord(r);
  });
}

function applySortMode() {
  const mode = $v('P7_SORT_MODE');

  if (mode === 'DATE') {
    sortSubtasksByDate();
    $('.subtask-number').addClass('subtask-number--disabled');
  } else {
    apex.region('SUBTASKS_IG').refresh();
    $('.subtask-number').removeClass('subtask-number--disabled');
  }
}
```
